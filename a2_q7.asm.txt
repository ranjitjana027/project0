.model small
.stack 100h
.data
	iter db 2
	num db ?
	result dw ?
	sum dw 0
.code
	main proc
	mov ax, @data
	mov ds, ax
	mov ax,1
	
	mov dx,1
	mov num,dl
	mov cx,0
	mov cl, iter
L0:	call fact
	inc num
	mov dx, result
	add sum, dx
	mov dx, sum
	add dx, 30h
	mov ah, 02h
	int 21h
	loop L0
	
	call fact
	mov dx, sum
	add dx, 30h
	mov ah, 02h
	int 21h
	mov ah, 4ch
	int 21h
	main endp



	fact proc
	mov ax, 1
	mov ch,0
	mov cl,num
L1:	mul cx
	loop L1
	mov result, ax
	
	ret
	fact endp
end main
